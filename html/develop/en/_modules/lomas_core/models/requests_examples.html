def make_dummy(example_query: Dict[str, JsonValue]) -> Dict[str, JsonValue]:

    example_query_dummy["dummy_nb_rows"] = DUMMY_NB_ROWS
    example_query_dummy["dummy_seed"] = DUMMY_SEED
